import * as React from "react";
import { cn } from "@/lib/utils";

interface GooeyTextMorphingProps {
  texts: string[];
  className?: string;
  duration?: number;
  pauseDuration?: number;
}

const GooeyTextMorphing = React.forwardRef(
  ({ texts, className, duration = 1.5, pauseDuration = 2 }, ref) => {
    const [index, setIndex] = React.useState(0);

    React.useEffect(() => {
      const interval = setInterval(() => {
        setIndex((prev) => (prev + 1) % texts.length);
      }, (duration + pauseDuration) * 1000);
      return () => clearInterval(interval);
    }, [texts, duration, pauseDuration]);

    return (
      

        
          
            {texts[index]}
          
        
        
      

    );
  },
);
GooeyTextMorphing.displayName = "GooeyTextMorphing";

export { GooeyTextMorphing };
